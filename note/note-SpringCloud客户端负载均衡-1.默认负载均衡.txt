1.使用步骤:注入RestTemplate时添加@LoadBalanced注解
    @Configuration
    public class ApplicationContextConfig {
        @Bean
        //必须加这个注解,不加这个注解访问会报错
        @LoadBalanced
        public RestTemplate getRestTemplate(){
            return new RestTemplate();
        }
    }
